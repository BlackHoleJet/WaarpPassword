76      public static void main(String[] args) throws Exception {
77          // TODO Auto-generated method stub
78          if (! WaarpPassword.loadOptions(args)) {
79              // Bad options
80              System.exit(2);
81          }
82          WaarpPassword waarpPassword = new WaarpPassword();
83          if (po == null && pi == null) {
84              // stop
85              System.err.println("Key written");
86              System.exit(0);
87          }
88          if (waarpPassword.clearPassword == null || waarpPassword.clearPassword.length() == 0) {
89              System.err.println("Password to crypt:");
90              String newp = waarpPassword.readString();
91              if (newp == null || newp.length() == 0) {
92                  System.err.println("No password as input");
93                  System.exit(4);
94              }
95              waarpPassword.setClearPassword(newp);
96              if (po != null) {
97                  waarpPassword.setPasswordFile(new File(po));
98                  waarpPassword.savePasswordFile();
99              }
100             if (clearPasswordView) {
101                 System.err.println("ClearPwd: "+waarpPassword.getClearPassword());
102                 System.err.println("CryptedPwd: "+waarpPassword.getCryptedPassword());
103             }
104         }
105     }

107     public static boolean loadOptions(String[] args) {
108         int i = 0;
109         if (args.length == 0) {
110             System.err.println(HELPOPTIONS);
111             return false;
112         }
113         for (i = 0; i < args.length; i++) {
114             if (args[i].equalsIgnoreCase("-ki")) {
115                 i++;
116                 if (i < args.length) {
117                     ki = args[i];
118                 } else {
119                     System.err.println("-ki needs a file as argument");
120                     return false;
121                 }
122             } else if (args[i].equalsIgnoreCase("-ko")) {
123                 i++;
124                 if (i < args.length) {
125                     ko = args[i];
126                 } else {
127                     System.err.println("-ko needs a file as argument");
128                     return false;
129                 }
130             } else if (args[i].equalsIgnoreCase("-pi")) {
131                 i++;
132                 if (i < args.length) {
133                     pi = args[i];
134                 } else {
135                     System.err.println("-pi needs a file as argument");
136                     return false;
137                 }
138             } else if (args[i].equalsIgnoreCase("-po")) {
139                 i++;
140                 if (i < args.length) {
141                     po = args[i];
142                 } else {
143                     System.err.println("-po needs a file as argument");
144                     return false;
145                 }
146             } else if (args[i].equalsIgnoreCase("-des")) {
147                 desModel = true;
148             } else if (args[i].equalsIgnoreCase("-blf")) {
149                 desModel = false;
150             } else if (args[i].equalsIgnoreCase("-pwd")) {
151                 i++;
152                 if (i < args.length) {
153                     pwd = args[i];
154                 } else {
155                     System.err.println("-pwd needs a password as argument");
156                     return false;
157                 }
158             } else if (args[i].equalsIgnoreCase("-cpwd")) {
159                 i++;
160                 if (i < args.length) {
161                     cpwd = args[i];
162                 } else {
163                     System.err.println("-cpwd needs a crypted password as argument");
164                     return false;
165                 }
166             } else if (args[i].equalsIgnoreCase("-clear")) {
167                 clearPasswordView = true;
168             } else {
169                 System.err.println("Unknown option: "+args[i]);
170                 return false;
171             }
172         }
173         if (ki == null && ko == null) {
174             System.err.println("You must specify one of ki or ko options");
175             return false;
176         }
177         if (ki == null) {
178             ki = ko;
179         }
180         if ((ki == null) && (po != null || pi != null)) {
181             System.err.println("If pi or po options are set, ki or ko options must be set also!\n");
182             return false;
183         }
184         if (pi == null && po == null && (pwd != null || cpwd != null)) {
185             System.err.println("Cannot create a password if no password GGP file is specified with pi or po options");
186             return false;
187         }
188         return true;
189     }

191     public WaarpPassword() throws Exception {
192         if (desModel) {
193             currentKey = new Des();
194         } else {
195             currentKey = new Blowfish();
196         }
197         if (ko != null) {
198             createNewKey();
199             saveKey(new File(ko));
200         }
201         if (ki != null) {
202             loadKey(new File(ki));
203         }
204         if (pi != null) {
205             setPasswordFile(new File(pi));
206             loadPasswordFile();
207         }
208         if (pwd != null) {
209             setClearPassword(pwd);
210         }
211         if (cpwd != null) {
212             setCryptedPassword(cpwd);
213         }
214         if (po != null) {
215             setPasswordFile(new File(po));
216             savePasswordFile();
217         }
218         if (clearPassword != null) {
219             if (clearPasswordView) {
220                 System.err.println("ClearPwd: "+getClearPassword());
221             }
222             System.err.println("CryptedPwd: "+getCryptedPassword());
223         }
224     }

335     /**
336      * Load the crypted password from the file
337      * @throws Exception
338      */
339     public void loadPasswordFile() throws Exception {
340         if (passwordFile.canRead()) {
341             int len = (int)passwordFile.length();
342             byte []key = new byte[len];
343             FileInputStream inputStream = null;
344             inputStream = new FileInputStream(passwordFile);
345             DataInputStream dis = new DataInputStream(inputStream);
346             dis.readFully(key);
347             dis.close();
348             setCryptedPassword(new String(key));
349         } else {
350             throw new CryptoException("Cannot read crypto file");
351         }
352     }
